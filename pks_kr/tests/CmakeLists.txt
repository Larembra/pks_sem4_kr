add_executable(pks_kr_unit_tests
        test_API.cpp
        test_nn.cpp
        test_player.cpp
        test_revolver.cpp
        scenario_status_flow.cpp
        scenario_revolver_cycle.cpp
        scenario_contestant_state.cpp
        scenario_player_status_variants.cpp
        scenario_revolver_printmagazine.cpp
        ../src/API/telegram.cpp
        ../src/secret/dotenv.cpp
        ../src/contestants/nn.cpp
        ../src/contestants/player.cpp
        ../src/contestants/contestant.cpp
        ../src/weapons/revolver.cpp
)

target_link_libraries(pks_kr_unit_tests PRIVATE
        Catch2::Catch2WithMain
)

find_package(CURL REQUIRED)
target_link_libraries(pks_kr_unit_tests PRIVATE
        ${CURL_LIBRARIES}
)
target_include_directories(pks_kr_unit_tests PRIVATE
        ${CURL_INCLUDE_DIRS}
)

if (WIN32)
    target_link_libraries(pks_kr_unit_tests PRIVATE ws2_32)
endif()

set_target_properties(pks_kr_unit_tests PROPERTIES
        RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/tests"
)

include(CTest)
include(Catch)
catch_discover_tests(pks_kr_unit_tests)
